\begin{tikzpicture}[remember picture,overlay]

    % White background
    \fill[white] (current page.south west) rectangle (current page.north east);
    
    % Simple black border
    \draw[black, line width=2pt] 
        ([xshift=20pt, yshift=20pt] current page.south west) rectangle 
        ([xshift=-20pt, yshift=-20pt] current page.north east);

    % Author name (top, black)
    \node[black, font=\Large, anchor=north, text width=\linewidth, align=center] 
        (author) at ([yshift=-80pt] current page.north) 
        {\textsc{\authorname}};

    % Book title (centered, large)
    \node[black, font=\Huge, anchor=north, text width=0.9\linewidth, align=center, below=20pt of author.south] 
        (title) {\textsc{\booktitle}};

    % Subtitle (if defined)
    \ifx\subtitle\undefined\else
        \if\relax\detokenize\expandafter{\subtitle}\relax\else
            \node[black, font=\large, text width=0.9\linewidth, align=center, anchor=north, below=10pt of title.south] 
                (subtitlenode) {\subtitle};
        \fi
    \fi

    % Simple Raspberry Pi logo representation (geometric)
    % Raspberry Pi board representation (larger and more detailed)
    \begin{scope}[remember picture, overlay, shift={(current page.center)}]
        % Main board rectangle (much larger)
        \draw[black, line width=2pt, fill=gray!15, rounded corners=2pt] 
            (-3, -2) rectangle (3, 2);
        
        % GPIO header (2x20 pins)
        \draw[black, line width=1pt, fill=black] 
            (-2.7, 1.2) rectangle (-1.7, 1.7);
        \foreach \x in {-2.6, -2.4, -2.2, -2.0, -1.8} {
            \foreach \y in {1.3, 1.6} {
                \fill[white] (\x, \y) circle (0.06);
            }
        }
        
        % Main processor (large square)
        \draw[black, line width=1pt, fill=gray!30] 
            (-0.5, -0.5) rectangle (0.5, 0.5);
        \node[black, font=\tiny] at (0, 0) {BCM};
        
        % USB ports (right side)
        \draw[black, line width=1pt, fill=gray!20] 
            (2.5, 0.8) rectangle (3.2, 1.3);
        \draw[black, line width=1pt, fill=gray!20] 
            (2.5, 0.2) rectangle (3.2, 0.7);
        
        % Ethernet port
        \draw[black, line width=1pt, fill=gray!20] 
            (2.5, -0.7) rectangle (3.2, -0.2);
        
        % Power connector (micro USB)
        \draw[black, line width=1pt, fill=black] 
            (-3.2, -0.3) rectangle (-2.8, 0.3);
        
        % HDMI port
        \draw[black, line width=1pt, fill=black] 
            (-3.2, 0.8) rectangle (-2.8, 1.3);
        
        % SD card slot
        \draw[black, line width=1pt, fill=gray!30] 
            (-1, -2.2) rectangle (1, -1.8);
        
        % Various smaller components
        \draw[black, line width=0.5pt, fill=gray!40] 
            (1.5, 0.8) rectangle (2.2, 1.2);
        \draw[black, line width=0.5pt, fill=gray!40] 
            (1.2, -1.5) rectangle (1.8, -1.0);
        \draw[black, line width=0.5pt, fill=gray!40] 
            (-1.8, 0.2) rectangle (-1.2, 0.6);
            
        % LEDs
        \fill[black] (2.3, 1.6) circle (0.08);
        \fill[black] (2.0, 1.6) circle (0.08);
        
        % Corner mounting holes
        \foreach \x/\y in {-2.7/-1.7, 2.7/-1.7, -2.7/1.7, 2.7/1.7} {
            \draw[black, line width=0.5pt] (\x, \y) circle (0.1);
        }
    \end{scope}

    % Publisher name at bottom
    \node[black, font=\large, anchor=south] 
        (publisher) at ([yshift=40pt] current page.south) 
        {\textsc{\publisher}};

\end{tikzpicture}
\end{center}
